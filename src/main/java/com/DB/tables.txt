create table products(
                         id serial primary key ,
                         code varchar(50) unique not null ,
                         name varchar(50) unique not null,
                         price float8 not null ,
                         url varchar(250)
);
create table user_statistics(
                                id serial primary key ,
                                login varchar(50) not null unique ,
                                email varchar(50) not null unique ,
                                storyName varchar(50) not null,
                                password varchar(50) not null
);
create table month_statistics(
                                 id serial primary key ,
                                 user_login varchar(50) references user_statistics(login),
                                 month_count int not null
);
create table product_amounts(
                                id serial primary key,
                                product_name varchar(50) references products(name) not null ,
                                month_statistic_id int references month_statistics(id) ,
                                amountInStock int default 0,
                                amountSold int default 0
);
